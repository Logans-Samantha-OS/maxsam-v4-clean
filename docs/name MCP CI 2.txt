name: MCP CI

on:
  pull_request:
  push:
    branches: [ "main" ]

jobs:
  mcp_ci:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Repo integrity checks
        run: |
          chmod +x scripts/validate_repo.sh
          scripts/validate_repo.sh

      - name: Validate n8n JSON (basic)
        run: |
          node scripts/validate_n8n_json.js

      - name: Validate directive references
        run: |
          node scripts/validate_directive_refs.js

      - name: Smoke test matrix (logical coverage)
        run: |
          node scripts/smoke_test_matrix.js
